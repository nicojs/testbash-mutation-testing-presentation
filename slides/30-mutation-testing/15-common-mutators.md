Original            | Mutated   
------------------|:-------
a + b | a - b
a / b | a * b   
a < b | a > b
a == b | a != b
a && b | a &#124;&#124; b
var drink = "Cola" | var drink = ""
var list = [1, 2, 3, 4] | var list = []
if (a > b) { ... } | if (true) { ... }
function fn() { ...} | function fn() { /* EMPTY */ }
